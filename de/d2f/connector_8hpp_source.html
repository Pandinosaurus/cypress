990  void generate_connections(size_t offs, size_t len, size_t subset_len,
991  size_t i0, size_t i1, const F &f) const
992  {
993  // Initialize a vector with the identity permutation
994  std::vector<size_t> perm(len);
995  for (size_t i = 0; i < len; i++) {
996  perm[i] = offs + i;
997  }
998 
999  // Iterate over each target neuron, create a new permutation containing
1000  // the source neuron indices and instantiate the connection
1001  std::uniform_int_distribution<size_t> distr(0, perm.size() - 1);
1002  for (size_t i = i0; i < i1; i++) {
1003  // Generate a new permutation for the first subset_len elements
1004  for (size_t j = 0; j < subset_len; j++) {
1005  std::swap(perm[j], perm[distr(*m_engine)]);
1006  }
1007 
1008  // Create the connections
1009  for (size_t j = 0; j < subset_len; j++) {
1010  f(i, perm[j]);
1011  }
1012  }
1013  }

1017  std::shared_ptr<RandomEngine> engine)
